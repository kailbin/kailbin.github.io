js:

	
/**
 * {
 *      tabTitle:'#tabTitle', //选项卡
 *      tabBody:'#tabBody',   // 选项卡体
 *      event:'click'         // 选项卡事件
 * }
 */
$.fn.simpleTabs = function (o) {
    o = o || {};
    var tabTitle = o.tabTitle || "#tabTitle";
    var tabBody = o.tabBody || "#tabBody";
    var event = o.event || "click";

    var $this = $(this);
    var $tabTitle = $this.find(tabTitle);
    var $tabBody = $this.find(tabBody);

    $tabBody.find("li").hide().first().show();
    $tabTitle.find("li").first().addClass("selected").end().bind(event, function () {
        $(this).addClass("selected").siblings("li").removeClass("selected");
        var index = $tabTitle.find("li").index(this);
        $tabBody.children().eq(index).show().siblings().hide();
    });
};
